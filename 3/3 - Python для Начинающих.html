import pygame

import random

import sys

import time

 

MAX_X = 3840

MAX_Y = 2160

MAX_SNOW = 300

SNOW_SIZE = 64

 

class Snow():

    def __init__(self, x, y):

        self.x = x

        self.y = y

        self.speed_y = random.randint(1, 3)

        self.speed_x = random.randint(-1, 1)

        self.img_num = random.randint(1, 4)

        self.image_filename = 'snow' + str(self.img_num) + '.png'

        self.image = pygame.image.load(self.image_filename).convert_alpha()

        self.image = pygame.transform.scale(self.image, (SNOW_SIZE, SNOW_SIZE))

 

    def move_snow(self):

        self.y += self.speed_y

        if self.y > MAX_Y:

            self.y = (0 - SNOW_SIZE)

        self.x += self.speed_x

        if self.x > MAX_X:

            self.x = (0 - SNOW_SIZE)

        elif self.x < (0 - SNOW_SIZE):

            self.x = MAX_X
